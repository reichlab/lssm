Simulated Data

Illustration of estimation and 1 step ahead prediction intervals.

y <- arima.sim(model = list(ar = c(.2, .5, .05), ma = .9), n = 200)

arma_fit <- fit_lssm(
  y = y,
  model = "arma",
  transformation = "none",
  p_ar = 3,
  q_ma = 1,
  verbose = FALSE)

forecasts <- predict(
  arma_fit,
  horizon = 1,
  forecast_representation = "quantile",
  quantile_levels = c(0.025, 0.25, 0.5, 0.75, 0.975))
